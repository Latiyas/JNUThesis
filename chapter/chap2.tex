\section{伪代码实例}
\begin{algorithm}
\caption{Algorithm 1}
\begin{algorithmic}[1] %[1] 能够显示行号
    \REQUIRE $n \geq 1$                  %输入条件
    \ENSURE $Sum = 1 + \cdots + n$       %输出
    \STATE $Sum \leftarrow 0$            %\STATE 命名演示
    \IF {$n < 1$}                        %条件语句
        \PRINT {Input Error}                 %打印语句
    \ELSE
        \FOR {$i = 0$ to n}          %FOR循环结构
            \STATE $Sum = Sum + i$\\
            \STATE $i = i + 1$
        \ENDFOR
    \ENDIF
    \RETURN Sum
\end{algorithmic}
\end{algorithm}
